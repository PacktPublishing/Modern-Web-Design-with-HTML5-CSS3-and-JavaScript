let val = prompt('what number?');
val = Number(val);
let num = 100;
let message = "nothing";
if (val > num) {
  message = `${val} was greater than ${num}`;
}
else if (val == num) {
  message = `${val} was equal to ${num}`;
}
else {
  message = `${val} is less than ${num}`;
}
document.getElementById('message').innerHTML = message;
